-- remove todos usuários que tem apenas café no arranchamento
UPDATE public.relatorios
SET usuarios_refeicoes = (
    SELECT jsonb_object_agg(key, value)
    FROM jsonb_each_text(usuarios_refeicoes)
    WHERE value <> 'C'
)
WHERE id = 108;



UPDATE relatorios
SET usuarios_refeicoes = (
    SELECT jsonb_object_agg(key, CASE WHEN value = '"CAJ"' THEN '"C"' ELSE value END)
    FROM jsonb_each(usuarios_refeicoes)
)
where data_relatorio = '2026-03-13'



update usuarios set padrao_semanal = '{
  "Sexta": "C",
  "Quarta": "CA",
  "Quinta": "CA",
  "Terça": "CA",
  "Domingo": "",
  "Segunda": "CA",
  "Sábado": ""
}' where id > 401 and id_om = 1 and id_patente = 1
